@app.route("/edit_appointment/", methods=["GET", "POST"]) def edit_appointment(appointment_id): conn = get_db_connection() cur = conn.cursor() if request.method == "POST": client_id = request.form["client_id"] service_id = request.form.get("service_id") or None appointment_date = request.form["appointment_date"] appointment_time = request.form["appointment_time"] duration_minutes = request.form.get("duration_minutes") or None room_number = request.form.get("room_number") status = request.form["status"] price_at_booking = request.form.get("price_at_booking") or None notes = request.form.get("notes") square_appointment_id = request.form.get("square_appointment_id") cur.execute(""" UPDATE appointments SET client_id = %s, service_id = %s, appointment_date = %s, appointment_time = %s, duration_minutes = %s, room_number = %s, status = %s, price_at_booking = %s, square_appointment_id = %s, notes = %s, updated_at = CURRENT_TIMESTAMP WHERE appointment_id = %s """, ( client_id, service_id, appointment_date, appointment_time, duration_minutes, room_number, status, price_at_booking, square_appointment_id, notes, appointment_id )) conn.commit() cur.close() conn.close() flash("Appointment updated successfully.", "success") return redirect(url_for("appointments")) cur.execute(""" SELECT appointment_id, client_id, service_id, appointment_date, appointment_time, duration_minutes, room_number, status, price_at_booking, square_appointment_id, notes FROM appointments WHERE appointment_id = %s """, (appointment_id,)) appointment = cur.fetchone() cur.execute(""" SELECT client_id, first_name, last_name FROM clients ORDER BY last_name, first_name """) clients = cur.fetchall() cur.execute(""" SELECT service_id, service_name FROM services ORDER BY service_name """) services = cur.fetchall() cur.close() conn.close() return render_template( "edit_appointment.html", appointment=appointment, clients=clients, services=services )
